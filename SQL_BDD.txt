CREATE TABLE civilite(
civ_id INT NOT NULL,
civ_lib VARCHAR(150) NOT NULL,
CONSTRAINT pk_civ PRIMARY KEY (civ_id) 
);


CREATE TABLE diplome(
dip_id INT NOT NULL AUTO_INCREMENT,
dip_lib VARCHAR(150) NOT NULL,
CONSTRAINT pk_dip PRIMARY KEY (dip_id)
);

CREATE TABLE contrat(
cont_id INT NOT NULL,
cont_ddb DATE NOT NULL,
cont_ddf DATE NULL,
cont_sal_brut DECIMAL(4,2) NOT NULL,
cont_charge DECIMAL(4,2) NOT NULL,
CONSTRAINT pk_cont PRIMARY KEY (cont_id)
);


CREATE TABLE conge(
conge_ddb DATE NOT NULL,
conge_ddf DATE NOT NULL,
CONSTRAINT pk_conge PRIMARY KEY (conge_ddb,conge_ddf)
);

CREATE TABLE service(
serv_id INT NOT NULL,
serv_lib VARCHAR(150) NOT NULL,
CONSTRAINT pk_serv PRIMARY KEY (serv_id)
);


CREATE TABLE personnel(
pers_id INT NOT NULL,
pers_nom VARCHAR(250) NOT NULL,
pers_prenom VARCHAR(250) NOT NULL,
pers_ddn DATE NOT NULL,
pers_adl1 VARCHAR(250) NOT NULL,
pers_adl2 VARCHAR(250) NULL,
pers_cp CHAR(5) NOT NULL,
pers_ville VARCHAR(150) NOT NULL,
pers_civ INT NOT NULL,
pers_cont INT NOT NULL,
pers_serv INT NOT NULL,
CONSTRAINT pk_pers PRIMARY KEY (pers_id)
);
